Implementation of Stack


class Stack:
 def __init__(self):
   self.stack = []


 def push(self,data):
   self.stack.append(data)


 def pop(self):
   if len(self.stack) == 0:
     return "Stack Underflow"
   return self.stack.pop()


 def peek(self):
   if len(self.stack) == 0:
     return "Stack Underflow"
   return self.stack[-1]


 def isEmpty(self):
   if len(self.stack) == 0:
     return True
   return False


 def size(self):
   return len(self.stack)
 
 def printStack(self):
   return self.stack




























Implementation of Queue


from collections import deque


class Queue:


 def __init__(self):
   self.queue = deque()


  def enqueue(self,data):
   self.queue.append(data)


  def isEmpty(self):
   if len(self.queue) == 0:
     return True
   return False


  def dequeue(self):
   if self.isEmpty() == True:
     return "Queue underflow"
   return self.queue.popleft()


  def peek(self):
   if self.isEmpty() == True:
     return "Queue underflow"
   return self.queue[0]


  def size(self):
   return len(self.queue)


  def printQueue(self):
   return self.queue
