Q.1 Merge Sort using Recursion
https://leetcode.com/problems/sort-an-array/


class Solution:
   def combineMergesort(self,left_sorted: List[int],right_sorted):
       res = []
       i = 0
       j = 0
       while i < len(left_sorted) and j < len(right_sorted):
           if left_sorted[i] <= right_sorted[j]:
               res.append(left_sorted[i])
               i += 1
           else:
               res.append(right_sorted[j])
               j += 1
       res.extend(left_sorted[i:])
       res.extend(right_sorted[j:])
       return res
   def mergeSort(self,arr,i,j):
       n = j - i + 1
       if n == 1:
           return [arr[i]]
       if n == 2:
           if arr[i] >= arr[j]:
               return [arr[j],arr[i]]
           else:
               return [arr[i],arr[j]]
       mid = i + (j-i) // 2
       left_sorted = self.mergeSort(arr,i,mid)


       right_sorted = self.mergeSort(arr,mid+1,j)


       return self.combineMergesort(left_sorted,right_sorted)
   def sortArray(self, nums: List[int]) -> List[int]:
       return self.mergeSort(nums,0,len(nums) - 1)
